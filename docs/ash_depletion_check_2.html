We test the ash-depletion model in Logolas on Yuichiâ€™s data. The data was collected from Alexandrov et al paper, where the mutation signatures were computed over around 70,000 somatic mutations. This is why we take \(nP/nQ=10^6\) as a default for this particular set up, and compare other choices of nP and nQ and interplay between them.

library(Logolas)

Data

mat=read.table(paste('../data/Fig4_rawdata/sig_',12,'.txt',sep = ''))
mat1=cbind(t(mat[2:3,1:4]),rep(NA,4),t(mat[4:5,1:4]))
rownames(mat1)=c('A','C','G','T')
colnames(mat1) = c("-2", "-1", "0", "1", "2")
mat2=cbind(rep(NA,6),rep(NA,6),t(mat[1,]),rep(NA,6),rep(NA,6))
colnames(mat2) = c("-2", "-1", "0", "1", "2")
rownames(mat2) = c("C>A", "C>G", "C>T", "T>A", "T>C", "T>G")
table = rbind(mat1, mat2)
